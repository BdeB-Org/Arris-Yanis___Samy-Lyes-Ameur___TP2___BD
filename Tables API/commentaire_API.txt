procedure ins (
p_ID_COM in COMMENTAIRE.ID_COM%type
,p_DATE_COM in COMMENTAIRE.DATE_COM%type
,p_TEXTE in COMMENTAIRE.TEXTE%type
,p_ARTICLE_ID_ARTICLE in COMMENTAIRE.ARTICLE_ID_ARTICLE%type
,p_UTILISATEUR_ID_USER in COMMENTAIRE.UTILISATEUR_ID_USER%type
) is
begin
insert into COMMENTAIRE(
ID_COM
,DATE_COM
,TEXTE
,ARTICLE_ID_ARTICLE

) values (
p_ID_COM
,p_DATE_COM
,p_TEXTE
,p_ARTICLE_ID_ARTICLE
,p_UTILISATEUR_ID_USER
);end;
-- update
procedure upd (
p_ID_COM in COMMENTAIRE.ID_COM%type
,p_DATE_COM in COMMENTAIRE.DATE_COM%type
,p_TEXTE in COMMENTAIRE.TEXTE%type
,p_ARTICLE_ID_ARTICLE in COMMENTAIRE.ARTICLE_ID_ARTICLE%type

,TEXTE = p_TEXTE
